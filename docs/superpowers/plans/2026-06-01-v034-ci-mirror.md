**Goal:** Add a GitHub Action that mirrors the v0.33 PMID-integrity hook server-side (closing the `--no-verify` bypass) plus an offline tests+build safety-net, and make `verify-pmids.mjs` CI-robust (API key, transient HTTP-error tolerance, network reporting).

// E3: final summary line for PMIDs left unverified by transient network errors.
export function networkSummaryLine(count) {
  return count > 0 ? `⚠️  ${count} PMID(s) unverified (network error)` : null
}

// E4: exit code decision. Blocking issues (MISMATCH/NOT_FOUND) → 1. In --strict,
// MAYBE_FP → 1. In --ci, leftover network-unverified PMIDs → 2 (workflow retries).
// Without --ci, network errors never block (hook path: exit 0). Blocking wins over ci/network.
export function computeExitCode({ blockingCount, maybeFpCount, networkErrCount, strictMode, ciMode }) {
  if (blockingCount > 0) return 1
  if (strictMode && maybeFpCount > 0) return 1
  if (ciMode && networkErrCount > 0) return 2
  return 0
}
```

- [ ] **Step 1: Create the workflow file** — `.github/workflows/ci.yml`:

```yaml
name: CI

on:
  push:
    branches: [master]
  pull_request:
    branches: [master]
  workflow_dispatch:

# Skip husky hook installation during `npm ci` (no local git hooks needed in CI).
env:
  HUSKY: 0

jobs:
  build-and-test:
    name: Build & Test (offline)
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
      - run: npm ci
      - name: Unit tests
        run: node --test scripts/test/*.test.mjs
      - name: Build
        run: npm run build

  pmid-gate:
    name: PMID integrity gate (NCBI)
    runs-on: ubuntu-latest
    env:
      NCBI_API_KEY: ${{ secrets.NCBI_API_KEY }}
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
      - run: npm ci
      - name: Verify PMIDs (lib-wide, 1 retry on transient)
        shell: bash
        run: |
          set +e
          node scripts/verify-pmids.mjs --batch --ci | tee verify-output.txt
          code=${PIPESTATUS[0]}
          if [ "$code" = "2" ]; then
            echo "::warning::Transient NCBI errors on first pass — retrying once"
            node scripts/verify-pmids.mjs --batch --ci | tee verify-output.txt
            code=${PIPESTATUS[0]}
          fi
          if [ "$code" = "1" ]; then
            echo "::error::PMID verification failed (fabricated/mismatched PMID)."
            exit 1
          fi
          exit 0
      - name: Job summary
        if: always()
        shell: bash
        run: |
          {
            echo "### PMID gate result"
            echo '```'
            tail -n 25 verify-output.txt 2>/dev/null || echo "(no output captured)"
            echo '```'
          } >> "$GITHUB_STEP_SUMMARY"
```
