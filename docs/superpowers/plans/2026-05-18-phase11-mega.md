```js
// After the existing webp write
const avifPath = inPath.replace(/\.png$/i, '.avif')
let avifStat = null
try { avifStat = await stat(avifPath) } catch {}
if (!avifStat || avifStat.mtimeMs < inStat.mtimeMs) {
  let avifPipeline = sharp(inPath)
  if (needsResize) {
    avifPipeline = avifPipeline.resize({
      width: RESIZE_THRESHOLD, height: RESIZE_THRESHOLD,
      fit: 'inside', withoutEnlargement: true,
    })
  }
  await avifPipeline.avif({ quality: AVIF_QUALITY, effort: AVIF_EFFORT }).toFile(avifPath)
  const newAvifStat = await stat(avifPath)
  const pct = ((1 - newAvifStat.size / inStat.size) * 100).toFixed(1)
  console.log(`  ${file}: AVIF ${(newAvifStat.size / 1024).toFixed(0)}kB (-${pct}%)`)
}
```

```jsx
  const [uiMap, setUiMap] = useState(UI_HU) // synchronous default
  const [loadingLang, setLoadingLang] = useState(false)

  useEffect(() => {
    sessionStorage.setItem('molekulax:lang', lang)
    if (lang === 'hu') {
      setUiMap(UI_HU)
      return
    }
    setLoadingLang(true)
    import(`./uiStrings-${lang}.js`)
      .then(mod => {
        const key = `UI_${lang.toUpperCase()}`
        setUiMap(mod[key])
        setLoadingLang(false)
      })
      .catch(err => {
        console.error('Failed to load locale', lang, err)
        setLoadingLang(false)
      })
  }, [lang])

  const t = (key) => uiMap[key] || key

  return (
    <LanguageContext.Provider value={{ lang, setLang, t, loadingLang }}>
      {children}
    </LanguageContext.Provider>
  )
}

export function useLang() {
  return useContext(LanguageContext)
}
```

### Task 5.3: Rewrite per-library `entries/index.js` + entries-meta

- [ ] **Step 1: Write `entries-meta-{hu,en,pl}.js` per library**

For each of the 4 libraries (peptides/nootropics/performance/pharmaceutical), create per-locale tile metadata file. Actually, since these are static, generate them via a script extension that walks all `<id>-<lang>.js` files and emits `entries-meta-<lang>.js`:

```js
// scripts/generate-entries-meta.mjs
import { readdir } from 'node:fs/promises'
import { writeFileSync } from 'node:fs'
import { join } from 'node:path'

const LIBS = ['peptides', 'nootropics', 'performance', 'pharmaceutical']
const LANGS = ['hu', 'en', 'pl']
const META_FIELDS = ['id', 'name', 'shortDesc', 'image', 'accentColor', 'tagColor', 'category', 'tier', 'wadaStatus']

for (const lib of LIBS) {
  for (const lang of LANGS) {
    const dir = `src/data/libraries/${lib}/entries`
    const files = (await readdir(dir)).filter(f => f.endsWith(`-${lang}.js`))
    const metas = []
    for (const f of files) {
      const mod = await import(`../${dir}/${f}`)
      const entry = mod.default
      const meta = {}
      for (const k of META_FIELDS) {
        if (entry[k] !== undefined) meta[k] = entry[k]
      }
      metas.push(meta)
    }
    writeFileSync(`src/data/libraries/${lib}/entries-meta-${lang}.js`, `export default ${JSON.stringify(metas, null, 2)}\n`)
    console.log(`Generated ${lib}/entries-meta-${lang}.js (${metas.length} entries)`)
  }
}
```
